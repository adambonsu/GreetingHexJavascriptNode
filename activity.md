Update app so it serves json

```js

app.get("/", (req, res) => {
  res.json("Greetings");
});

app.listen(3000, () => {
  console.log("Server is now running");
});
```

Serve the app from `server.js`


Update `app.js`

```js
const express = require("express");
const app = express();

app.get("/", (req, res) => {
  res.send("Hello World!");
});

module.exports = app;
```

Serve app

```bash
node server.js
```
